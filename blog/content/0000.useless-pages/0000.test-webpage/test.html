C++
我们可以使用一个空指针NULL来测试OS中是否支持system()函数： 

                    
                        #include<iostream>
                        #include<cstdlib>  // 引入system函数

                        int main() {
                            if (system(NULL)) {
                                std::cout<<"支持";
                            } else {
                                std::cout<<"不支持";
                            }
                            return 0;
                        }
