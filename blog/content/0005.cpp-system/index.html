C++ 中的 system 函数

在 C++ 中，system 函数可在 Windows、Unix 等系统上有效，其用于执行系统命令，可以使用 stdlib.h、cstdlib 等声明。此函数原型如下：

                    
                        int system(const char *command);

它的作用是执行一个命令行命令，command 是一个指向命令行命令字符串的指针。这里有一些简单的示例：

  1. 清屏 
                                
                                system("cls");    // Windows
                                system("clear");  // Unix
  2. 查看当前目录文件 
                                
                                system("dir");    // Windows
                                system("ls -l");  // Unix
  3. 删除文件 
                                
                                system("del file.txt");  // Windows
                                system("rm file.txt");   // Unix

常用的还有 system("pause") 等等。但是使用时要多加考虑，因为执行该命令时有以下繁重的步骤：

  • 挂起程序，调用操作系统打开系统 shell。
  • 操作系统找到等待执行的命令，并分配内存以执行命令。
  • 重新分配内存，退出 shell 并恢复程序。

我们可以使用一个空指针 NULL 来测试 OS 中是否支持 system() 函数： 

                        
                            #include<iostream>
                            #include<cstdlib>

                            int main() {
                                if (system(NULL)) {
                                    std::cout<<"支持";
                                } else {
                                    std::cout<<"不支持";
                                }
                                return 0;
                            }

接下来我们以Windows系统为例，通过一些示例体现该函数的用法。

这是一个实现定时关机的程序： 

                    
                        #include<iostream>
                        #include<cstdlib>
                        using namespace std;

                        int main() {
                            int seconds;
                            cout<<"-----定时关机设定-----"<<endl<<"[输入-1以取消关机计划]"<<endl<<"请输入关机秒数：";
                            cin>>seconds;
                            if (seconds==-1) {
                                int exit_status=system("shutdown -a");
                                if (exit_status==1116) {
                                    cout<<"[无可用关机计划]";
                                    return 1;
                                } else {
                                    cout<<"[已取消关机计划]";
                                    return 0;
                                }
                            } else if (seconds<0) {
                                cout<<"[输入的数字无效]"<<endl;
                                return 1;
                            }
                            string command="shutdown -s -t "+to_string(seconds);
                            int exit_status=system(command.c_str());
                            if (exit_status==0) {
                                cout<<"将在"<<seconds<<"秒后关机"<<endl;
                            } else {
                                cout<<"关机命令执行过程中出现错误，退出状态码："<<exit_status<<endl;
                            }
                            return 0;
                        }

这个程序让用户输入一个时间，然后等待该时间后关机。由于使用到了to_string()函数，所以如果编译失败，请在编译选项中加入-std=c++11即可正常编译。
当然，如果只需要一个简单的关机功能，我们可以删去大部分的代码，如下所示： 

                    
                        #include<iostream>
                        #include<cstdlib>
                        using namespace std;

                        int main() {
                            int seconds;
                            cout<<"请输入关机秒数：";
                            cin>>seconds;
                            system(("shutdown -s -t "+to_string(seconds)).c_str());
                            cout<<"将在"<<seconds<<"秒后关机"<< endl;
                            return 0;
                        }

下面是一个仿Wordle的小游戏，不过使用了数字代替，这里我们用system("cls")达到清屏效果，保证游戏体验良好。 

                    
                        #include<bits/stdc++.h>
                        #define line "----------------"
                        using namespace std;

                        void print(int i) {
                            cout<<"Guess the number"<<endl<<" within 5 times"<<endl<<"["<<5-i-1<<" chances rest]"<<endl<<line<<endl;
                        }

                        int main() {
                            char answer[5];
                            srand(time(0));
                            for (int i=0;i<5;i++) {
                                answer[i]=rand()%10+48;
                            }
                            // cout<<answer<<endl;
                            char userlog[5][5];
                            print(-1);
                            for (int i=0;i<5;i++) {
                                cout<<"     ";
                                for (int j=0;j<5;j++) {
                                    cin>>userlog[i][j];
                                }
                                system("cls");
                                int count=0;
                                print(i);
                                for (int k=0;k<i+1;k++) {
                                    cout<<"     ";
                                    for (int j=0;j<5;j++) {
                                        if (userlog[k][j]==answer[j]) {
                                            printf("\x1b[1;32m");
                                            cout<<userlog[k][j];
                                            if (k==i) count++;
                                            printf("\x1b[0m");
                                        } else if (find(answer,answer+5,userlog[k][j])!=answer+5) {
                                            printf("\x1b[1;33m");
                                            cout<<userlog[k][j];
                                            printf("\x1b[0m");
                                        } else if (userlog[k][j]!=answer[j]) {
                                            printf("\x1b[1;31m");
                                            cout<<userlog[k][j];
                                            printf("\x1b[0m");
                                        }
                                        if (count==5) {
                                            cout<<endl<<line<<endl<<"You won!"<<endl<<"Used chances: "<<i+1;
                                            return 0;
                                        }
                                    }
                                    cout<<endl;
                                }
                            }
                            cout<<line<<endl<<"You lost!"<<endl<<"The answer is: "<<answer;
                            return 0;
                        }

这其中的printf("\x1b[1;32m")等用于调节字符颜色，注意不能使用system("color [attr]")来代替，因为该命令会切换终端所有文本的颜色。
不过在一些设备上可能无法使用上述转义字符更改颜色，我们可以用SetConsoleTextAttribute()函数来代替。

                            }
                            // cout<<answer<<endl;
                            char userlog[5][5];
                            setcolor(FOREGROUND_RED|FOREGROUND_GREEN|FOREGROUND_BLUE);
                            print(-1);
                            for (int i=0;i<5;i++) {
                                cout<<"     ";
                                for (int j=0;j<5;j++) {
                                    cin>>userlog[i][j];
                                }
                                system("cls");
                                int count=0;
                                print(i);
                                for (int k=0;k<i+1;k++) {
                                    cout<<"     ";
                                    for (int j=0;j<5;j++) {
                                        if (userlog[k][j]==answer[j]) {
                                            setcolor(FOREGROUND_GREEN);
                                            cout<<userlog[k][j];
                                            if (k==i) count++;
                                            setcolor(FOREGROUND_RED|FOREGROUND_GREEN|FOREGROUND_BLUE);
                                        } else if (find(answer,answer+5,userlog[k][j])!=answer+5) {
                                            setcolor(FOREGROUND_RED|FOREGROUND_GREEN);
                                            cout<<userlog[k][j];
                                            setcolor(FOREGROUND_RED|FOREGROUND_GREEN|FOREGROUND_BLUE);
                                        } else if (userlog[k][j]!=answer[j]) {
                                            setcolor(FOREGROUND_RED);
                                            cout<<userlog[k][j];
                                            setcolor(FOREGROUND_RED|FOREGROUND_GREEN|FOREGROUND_BLUE);
                                        }
                                        if (count==5) {
                                            cout<<endl<<line<<endl<<"You won!"<<endl<<"Used chances: "<<i+1;
                                            return 0;
                                        }
                                    }
                                    cout<<endl;
                                }
                            }
                            cout<<line<<endl<<"You lost!"<<endl<<"The answer is: "<<answer;
                            return 0;
                        }

这是一个修改后的版本，与Wordle原游戏效果几乎相同，只是猜测时使用的是小写字母。 

                    
                        #include<bits/stdc++.h>
                        #define streak "----------------"
                        using namespace std;
                            
                        char usertemp[5],userlog[6][5];

                        void print(int i) {
                            cout<<"  Guess a word"<<endl<<" within 6 times"<<endl<<"["<<6-i-1<<" chances rest]"<<endl<<streak<<endl;
                        }

                        int main() {
                            ifstream file("words.txt");
                            vector<string> lines;
                            string line;
                            srand(time(0));
                            int temp=rand()%10000;
                            while (file>>line) {
                                lines.push_back(line);
                            }
                            file.close();
                            srand(time(0));
                            int index=(rand()%10000+temp)/2;
                            mt19937 rng(static_cast<unsigned int>(time(nullptr)));
                            shuffle(lines.begin(),lines.end(),rng);
                            string word=lines[index];
                            char answer[5];
                            index=0;
                            for (char c:word) {
                                answer[index++]=c;
                            }
                            // cout<<answer<<endl;
                            print(-1);
                            int i=0,exist;
                            while (i<6) {
                                cout<<"     ";
                                for (int j=0;j<5;j++) {
                                    cin>>usertemp[j];
                                }
                                system("cls");
                                int count=0;
                                string userans;
                                for (const auto& letter:usertemp) {
                                    userans+=letter;
                                }
                                auto it=find(lines.begin(),lines.end(),userans);
                                if (it!=lines.end()) {
                                    exist=1;
                                    print(i);
                                    for (int j=0;j<5;j++) {
                                        userlog[i][j]=usertemp[j];
                                    }
                                } else {
                                    exist=0;
                                    print(i-1);
                                }
                                for (int k=0;k<i+1;k++) {
                                    cout<<"     ";
                                    for (int j=0;j<5;j++) {
                                        if (userlog[k][j]==answer[j]) {
                                            printf("\x1b[1;32m");
                                            cout<<userlog[k][j];
                                            if (k==i) count++;
                                            printf("\x1b[0m");
                                        } else if (find(answer,answer+5,userlog[k][j])!=answer+5) {
                                            printf("\x1b[1;33m");
                                            cout<<userlog[k][j];
                                            printf("\x1b[0m");
                                        } else if (userlog[k][j]!=answer[j]) {
                                            printf("\x1b[1;31m");
                                            cout<<userlog[k][j];
                                            printf("\x1b[0m");
                                        }
                                        if (count==5) {
                                            cout<<endl<<streak<<endl<<"You won!"<<endl<<"Used chances: "<<i+1;
                                            return 0;
                                        }
                                    }
                                    if (exist==0&&k==i) {
                                        continue;
                                    } else {
                                        cout<<endl;
                                    }
                                }
                                if (exist==1) {
                                    i++;
                                } else {
                                    printf("\x1b[1;31m");
                                    cout<<userans<<"  <-- Not in word list"<<endl;
                                    printf("\x1b[0m");
                                }
                            }
                            cout<<streak<<endl<<"You lost!"<<endl<<"The answer is: "<<answer;
                            return 0;
                        }
